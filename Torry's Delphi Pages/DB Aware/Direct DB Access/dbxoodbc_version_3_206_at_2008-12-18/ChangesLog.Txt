  Example of connection to oterro:
  ....
  procedure OterroConnect(SQLConnection: TSQLConnection;
    const dbName, UserName, Password: string;
    iOterroVersion: integer = 3;
    bDirectOdbc: Boolean = True;
    bUseAlternativeDrivers: Boolean = True;
    bLoginPrompt: Boolean = False
  );
  begin
     SQLConnection.Close;

       SQLConnection.DriverName    := 'OterroOdbcDriver';
       SQLConnection.GetDriverFunc := 'getSQLDriverODBC';
       SQLConnection.LibraryName   := 'dbxoodbc.dll';

       SQLConnection.Params.Clear;

       SQLConnection.Params.Values['User_Name'] := UserName;
       SQLConnection.Params.Values['Password']  := Password;
       SQLConnection.LoginPrompt := bLoginPrompt;

       if bDirectOdbc then
       begin
         if iOterroVersion < 3 then
           SQLConnection.VendorLib     := '2:ot2k_32.dll'
         else
           SQLConnection.VendorLib     := 'oterro.dll';
         if bUseAlternativeDrivers then
         begin
           if iOterroVersion < 3 then
             SQLConnection.VendorLib     := SQLConnection.VendorLib + ';oterro.dll'
           else
             SQLConnection.VendorLib     := SQLConnection.VendorLib + ';2:ot2k_32.dll';
         end;
       end
       else
       begin
         SQLConnection.VendorLib     := '2:odbc32.dll';
         if bUseAlternativeDrivers then
         begin
           if iOterroVersion < 3 then
             SQLConnection.VendorLib     := SQLConnection.VendorLib + ';2:ot2k_32.dll;oterro.dll'
           else
             SQLConnection.VendorLib     := SQLConnection.VendorLib + ';oterro.dll;2:ot2k_32.dll'
         end;
         if iOterroVersion < 3 then
           SQLConnection.Params.Values['Database']  := 'DRIVER={Oterro Database Driver (*.rb1)};'
         else
           SQLConnection.Params.Values['Database']  := 'DRIVER={Oterro 3.0 Database Driver (*.rb1)};';
       end;

       SQLConnection.Params.Values['Database']  := SQLConnection.Params.Values['Database'] +
         'DBQ=' + dbName;

       SQLConnection.Params.Values['Custom String'] := 'coCatPrefix=DBQ';
       SQLConnection.Params.Values['Trim Char'] := 'True';

     SQLConnection.Open;
  end;
  ....
